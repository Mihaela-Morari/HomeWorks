/*1. Пользователь вводит год своего рождения, вывести
информацию о возрасте пользователя в разные года
до текущего. Например, вводим «2003». Вывод: «в
2003 было 1 год», «в 2004 было 2 год», «в 2005 было
3 год» и т.д.*/

let userYear = prompt('Enter the year you were born!');
if(!isNaN(userYear) && userYear<=2023){
    userYear=Number(userYear);
    console.log(`In ${userYear} you were 1 year old`);
    let age=1;
    for(let i=userYear+1;i<2023;i++){
        age++;
        console.log(`In ${i} you were ${age} years old`)
    }
    console.log(`In 2023 you are ${2023-userYear+1} year old`);
    }else{
        alert('You entered wrong!');
    }
